235  // Send batch of transactions to backups
236  auto start_time = std::chrono::steady_clock::now();
237  int status = KVCG_ESUCCESS;
238  std::set<Server*> backedUpToList;
239  std::set<K> testKeys(keys.begin(), keys.end());
240 
241  cse498::unique_buf check(1);
242  cse498::unique_buf rawBuf;
243 
244  // Validate lengths match of keys/values
245  if (keys.size() != values.size()) {
246  LOG(ERROR) << "Attempting to log differing number of keys and values";
247  status = KVCG_EINVALID;
248  goto exit;
249  }
250 
251  // TBD: Should we support duplicate keys?
252  // Warn for now
253  if (testKeys.size() != keys.size()) {
254  LOG(WARNING) << "Duplicate keys being logged, assuming vector is ordered";
255  //LOG(ERROR) << "Can not log put with duplicate keys!";
256  //status = KVCG_EINVALID;
257  //goto exit;
258  }
259 
260  // FIXME: Investigate/fix race condition if multiple clients
261  // issue put at same time.
262  // Asynchronously send to all backups, check for success later
263  for (auto tup : boost::combine(keys, values)) {
264  K key;
265  V value;
266  boost::tie(key, value) = tup;
267 
268  LOG(INFO) << "Logging PUT (" << key << "): " << value;
269  BackupPacket<K,V>* pkt = new BackupPacket<K,V>(key, value);
270 
271  auto elem = this->logged_puts->find(key);
272  if (elem == this->logged_puts->end()) {
273  //LOG(DEBUG4) << "Recording having logged key " << key;
274  this->logged_puts->insert({key, pkt});
275  } else {
276  //LOG(DEBUG4) << "Replacing log entry for self key " << pkt->getKey() << ": " << elem->second->getValue() << "->" << pkt->getValue();
277  elem->second = pkt;
278  }
279 
280  char* rawData = pkt->serialize();
281  size_t dataSize = pkt->getPacketSize();
282  LOG(DEBUG4) << "raw data: " << (void*)rawData;
283  LOG(DEBUG4) << "data size: " << dataSize;
284  rawBuf.cpyTo(rawData, dataSize);
285  uint64_t checkKey = 7;
286 
287  if (dataSize > MAX_LOG_SIZE) {
288  LOG(ERROR) << "Can not log data size (" << dataSize << ") > " << MAX_LOG_SIZE;
289  status = KVCG_EINVALID;
290  goto exit;
291  }
292 
293  for (auto backup : backupServers) {
294  if (!backup->isBackup(key)) {
295  LOG(DEBUG2) << "Skipping backup to server " << backup->getName() << " not tracking key " << key;
296  continue;
297  }
298 
299  if (backup->alive) {
300  LOG(DEBUG) << "Backing up to " << backup->getName();
301  // FIXME: Only do once, not for every key
302  backup->backup_conn->register_mr(
303  check,
304  FI_SEND | FI_RECV | FI_WRITE | FI_REMOTE_WRITE | FI_READ | FI_REMOTE_READ,
305  checkKey);
306 
307  backup->backup_conn->register_mr(
308  rawBuf,
309  FI_SEND | FI_RECV | FI_WRITE | FI_REMOTE_WRITE | FI_READ | FI_REMOTE_READ,
310 #ifdef FT_ONE_SIDED_LOGGING
311  this->logging_mr_key);
312 #else
313  (uint64_t) 8);
314 #endif // FT_ONE_SIDED_LOGGING
315 
316 
317 #ifdef FT_ONE_SIDED_LOGGING
318  do {
319  // TODO: Make more parallel, right now do not right to backup mr
320  // if it has not read previous write
321  backup->backup_conn->read(check, 1, 0, this->logging_mr_key);
322  } while (check.get()[0] != '\0');
323 
324  // TBD: Ask network-layer for async write
325  backup->backup_conn->write(rawBuf, dataSize, 0, this->logging_mr_key);
326 #else
327  // FIXME: use async_send... but does not work for some reason
328  //backup->backup_conn->async_send(rawBuf, dataSize);
329  backup->backup_conn->send(rawBuf, dataSize);
330  backedUpToList.insert(backup);
331 #endif
332  } else {
333  LOG(DEBUG2) << "Skipping backup to down server " << backup->getName();
334  }
335  }
336  }
337 
338 #ifndef FT_ONE_SIDED_LOGGING
339  // Wait for all sends to complete
340  for (auto backup : backedUpToList) {
341  LOG(DEBUG2) << "Waiting for sends to complete on " << backup->getName();
342  backup->backup_conn->wait_for_sends();
343  }
344 #endif
